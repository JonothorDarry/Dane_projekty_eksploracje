#vif calculation: https://towardsdatascience.com/targeting-multicollinearity-with-python-3bd3b4088d0b
from statsmodels.stats.outliers_influence import variance_inflation_factor
from statsmodels.tools.tools import add_constant

def compute_vif(df, considered_features):
    proto_df = df[considered_features].copy()
    proto_df = proto_df.dropna()
    print(f'Length without removed (nans): {len(proto_df)}')
    
    X = proto_df[considered_features]
    X['intercept'] = 1
    
    vif = pd.DataFrame()
    vif["Variable"] = X.columns
    vif["VIF"] = [variance_inflation_factor(X.values, i) for i in range(X.shape[1])]
    vif = vif[vif['Variable']!='intercept']
    return vif

def blend_df_pca(df, neue_df, features):
    indexes = []
    for i in range(df.shape[0]):
        if math.isnan(df[features[0]][i]):
            indexes.append(i)            
    
    for label in neue_df:
        proper_list = []
        used = 0
        for i in range(df.shape[0]):
            if i not in indexes:
                proper_list.append(neue_df[label][i-used])
            else:
                proper_list.append(None)
                used += 1
        df[label] = proper_list
    
    remove_meaningless(df, features)

feature_groups = [
    {'features':['rgb_red_std_dev', 'rgb_red_mean', 'rgb_green_std_dev',
       'rgb_green_mean', 'rgb_blue_std_dev', 'rgb_blue_mean',
       'hsv_hue_std_dev', 'hsv_saturation_std_dev',
       'hsv_saturation_mean', 'hsv_value_std_dev', 'hsv_value_mean'], 
    'components':3, 'name':'color'},
    
    {'features':['view_count', 'likes', 'dislikes', 'comment_count'], 'components':2, 'name':'interpersonal'},
    {'features':['screams', 'screams_to_normal_proportion'], 'components':1, 'name':'screamy'}
]

for system in feature_groups:
    features = system['features']
    n_components = system['components']
    naming = system['name']
    
    proto_df = df[features].dropna()
    x = proto_df.loc[:, features].values
    x = StandardScaler().fit_transform(x)
    
    pca = PCA(n_components=n_components)
    principalComponents = pca.fit_transform(x)

    principalDf = pd.DataFrame(data = principalComponents, columns = [f'{naming}_{x}' for x in range(1, 1+n_components)])
    blend_df_pca(df, principalDf, features)
